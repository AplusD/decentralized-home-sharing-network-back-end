CREATE FUNCTION GET_NEW_HOUSE_ID() RETURNS INT DETERMINISTIC
		BEGIN 
			DECLARE newId INT DEFAULT 0;
			SELECT COUNT(*) INTO newId FROM `Houses`;
			RETURN newId;
		END 
		
CREATE FUNCTION GET_NEW_ACCOUNT_ID() RETURNS INT DETERMINISTIC
		BEGIN 
			DECLARE newId INT DEFAULT 0;
			SELECT COUNT(*) INTO newId FROM `Users`;
			RETURN newId;
		END
